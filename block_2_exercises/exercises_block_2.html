Part 1: Deep Feedforward Networks¶

  • Complete the following class for defining a flexible deep binary classification model using TensorFlow.
  • The argument layersizes will be a list of integers specifying the number of units per layer, starting with the input layer (dimensionality of the data) and potentially several hidden layers. The output layer will not be part of this list. E.g., [ 128, 64 ] means the data is 128-dimensional and there shall be one hidden layer with 64 neurons.
  • All hidden layers shall use the ReLU activation function tf.nn.relu. The output layer should use the sigmoid function tf.sigmoid.
  • Use tf.nn.sigmoid_cross_entropy_with_logits as loss.
  • optimizer will be one of these strings: gd for gradient descent, mom for Momentum, adam for Adam, rms for RMSProp. Use the respective optimzers from https://www.tensorflow.org/api_guides/python/train#Optimizers to minimize the cost.
  • learning_rate will be a float specifiying the learning rate.
  • It is recommended to keep the convention of having the input data as row vectors stacked together vertically, when you hand the data to the model via placeholders. This way you can stick to the formulas we have seen several times:

$$\large \begin{array}{rl} A^{[0]} & = X\\ z^{[1]} & = A^{[0]} W^{[1]} + b^{[1]}\\ A^{[1]} & = r(z^{[1]})\\ \\ z^{[\ell]} & = A^{[\ell-1]} W^{[\ell]} + b^{[\ell]}\\ A^{[\ell]} & = g^{[\ell]}(z^{[\ell]})\\ \\ \hat{Y} & = A^{[L]} \end{array} $$

class DeepBinaryClassificationModel():
    def __init__(self, layersizes, optimizer, learning_rate, momentum=0.01):
        tf.set_random_seed(SEED)
        inputDimensions = layersizes[0]
        
        # set up placeholders for the input
        self.inputs = tf.placeholder(tf.float32, shape=[None, inputDimensions], name='inputs')
        self.outputs = tf.placeholder(tf.float32, shape=[None, 1], name='outputs')
        
        # set up the hidden layers (ReLU) in a loop
        self.W = [None] * len(layersizes)
        self.b = [None] * len(layersizes)
        self.z = [None] * len(layersizes)
        self.a = [None] * len(layersizes)
        
        self.a[0] = self.inputs


        for layerIdx, neuronCount in enumerate(layersizes):
            if layerIdx == 0:
                continue;
    
            previousLayerIdx = layerIdx - 1
            
            self.W[layerIdx] = tf.Variable(tf.random_normal([layersizes[previousLayerIdx], neuronCount]))
            self.b[layerIdx] = tf.Variable(tf.random_normal([1, neuronCount]))           
            self.z[layerIdx] = tf.add(tf.matmul(self.a[previousLayerIdx], self.W[layerIdx]), self.b[layerIdx])
            self.a[layerIdx] = tf.nn.relu(self.z[layerIdx])


        # set up the final layer (sigmoid)
        self.WOut = tf.Variable(tf.random_normal([layersizes[-1], 1]))
        self.bOut = tf.Variable(tf.random_normal([1, 1]))
        self.zOut = tf.add(tf.matmul(self.a[-1], self.WOut), self.bOut)
        self.yhat = tf.sigmoid(self.zOut, name='yhat')

        # set up the cost and the training step
        self.cost = tf.reduce_mean(tf.nn.sigmoid_cross_entropy_with_logits(labels=self.outputs, logits=self.zOut), name='cost')
        optimizers = {
            "gd": tf.train.GradientDescentOptimizer(learning_rate).minimize(self.cost),
            "mom": tf.train.MomentumOptimizer(learning_rate, momentum).minimize(self.cost),
            "adam": tf.train.AdamOptimizer(learning_rate).minimize(self.cost),
            "rms": tf.train.RMSPropOptimizer(learning_rate).minimize(self.cost)
        }
        self.train_step = optimizers[optimizer]
        
        self.accuracy = tf.contrib.metrics.accuracy(
            tf.cast(self.outputs, tf.bool),
            tf.cast(tf.round(self.yhat), tf.bool),
            name='accuracy_computation')
        
        tf.summary.scalar('Accuracy', self.accuracy)
        tf.summary.scalar('Cost', self.cost)

        self.merged = tf.summary.merge_all()

def test_model(layersizes, optimizer, learning_rate, epochs):
    tf.reset_default_graph() # Otherwise an InvalidArgument exception occurs when calculating the summary.

    # instantiate your model
    model = DeepBinaryClassificationModel(layersizes, optimizer, learning_rate)
    init = tf.global_variables_initializer()

    # start a TensorFlow session, repeatedly train your network in an epoch loop
    with tf.Session() as sess:
        sess.run(init)

        layerConfiguration = '-'.join(map(str, layersizes))
        configuration = f'{layerConfiguration}_{optimizer}_{learning_rate}_{epochs}'
        nowstr = datetime.datetime.now().strftime('%Y-%m-%d_%H.%M.%S')
        trainwriter = tf.summary.FileWriter(f'summaries_hidden_layer_more/{nowstr}_train_{configuration}', sess.graph)
        testwriter = tf.summary.FileWriter(f'summaries_hidden_layer_more/{nowstr}_test_{configuration}')

        for epoch in range(1, epochs + 1):
            fd_train = { model.inputs: X_train, model.outputs: y_train.reshape(-1, 1) }
            fd_test = { model.inputs: X_test, model.outputs: y_test.reshape(-1, 1) }

            _, cost = sess.run([model.train_step, model.cost], feed_dict=fd_train)

            train_probas = sess.run(model.yhat, feed_dict=fd_train)
            test_probas = sess.run(model.yhat, feed_dict=fd_test)
            if epoch % 500 == 0:
                train_acc, summary = sess.run([ model.accuracy, model.merged ], feed_dict=fd_train)
                trainwriter.add_summary(summary, epoch)
                trainwriter.flush()

                # the same using the test data
                test_acc, summary = sess.run([ model.accuracy, model.merged ], feed_dict=fd_test)
                testwriter.add_summary(summary, epoch)
                testwriter.flush()


                print('Epoch %4d of %d Train accuracy: %f Test accuracy: %f' % (epoch, epochs, train_acc, test_acc))

Part 2: Convolutional Networks¶

Part 2.1: Implement a CNN classification model¶

Implement the deep CNN model which we discussed in class, as shown in the picture below. Note, however, that the picture shows the situation of processing one single image at a time. We want to process images in batches, so all the sizes need an additional number in the front. Let's assume a batch size of 100 for the description in the following, and an image size of 32 x 32 pixels and 3 color channels. In your implementation, all of these will be parameters passed to the consturctor of the model.

  1. The inputs shall be of shape 100 x 32 x 32 x 3 (batch_size x image_height x image_width x image_channels).
  2. The outputs shall be of shape 100 x 1 (batch_size x 1).
  3. Use tf.placeholders for the inputs and outputs, and add an additional placeholder num for the number of inputs (we want to be able to pass in fewer than batch_size examples).
  4. Next we want a convolutional layer with filter size 5 x 5, a stride of 1, no padding (padding='VALID'), and 6 filters. You can define the whole filter bank of six filters at once, as a tf.Variable of shape 5, 5, 3, 6.
  5. Use the function tf.nn.conv2d for the convolution.
  6. Add biases, and use the tf.nn.relu function as activation.
  7. Next, apply max pooling using the tf.nn.max_pool function, with ksize=[ 1, 2, 2, 1 ] and strides=[ 1, 2, 2, 1 ] and padding='VALID' (= no padding).
  8. Add another convolutional layer with filter size 5 x 5, a stride of 1, no padding, and 16 filters.
  9. Add biases and again use the ReLU function.
  10. Reshape the result to (batchsize, -1) -- in our example, 100 x 400 -- using the function tf.reshape.
  11. Add a fully connected layer of size 120, using the ReLU function, just like we did in non-convolutional networks.
  12. Add another fully connected layer of size 84, using the ReLU function.
  13. For the loss, use tf.reduce_sum(tf.nn.sigmoid_cross_entropy_with_logits(...)) (we used tf.reduce_mean in the past) and divide the result by tf.cast(self.num, tf.float32). This way, we can compute the average loss correctly even when we don't have a complete batch.
  14. Use the Adam optimizer
  15. Use tf.sigmoid as final activation
  16. Don't compute the accuracy for the current batch, but instead compute the number of correct predictions (we will compute the accuracy across all batches outside): 
    equal = tf.equal(
 tf.cast(self.output[0:self.num], tf.bool),
 tf.cast(tf.round(self.prediction[0:self.num]), tf.bool),
)
self.num_correct = tf.reduce_sum(tf.cast(equal, tf.float32), name='accuracy_computation')
  17. You don't need to add any summaries, we will also do that outside.

CONV1_f = 5

CONV2_n = 16
FULL1_n = 120
FULL2_n = 84

class CNN:
    def __init__(self, num_classes, image_height, image_width, image_channels, batch_size, learning_rate):
        
        # Parameters
        self.batch_size = batch_size
        self.learning_rate = learning_rate
        self.image_height = image_height
        self.image_width = image_width
        self.image_channels = image_channels
        
        # Inputs
        self.num = tf.placeholder(tf.int32)
        self.input = tf.placeholder(tf.float32, shape=[batch_size, image_height, image_width, image_channels])
        self.output = tf.placeholder(tf.float32, shape=[None, 1])
        
        # Weights & biases
        self.Wconv1 = tf.Variable(tf.random_normal([CONV1_f, CONV1_f, image_channels, CONV1_n]))
        self.bconv1 = tf.Variable(tf.random_normal([CONV1_n]))
        
        self.Wconv2 = tf.Variable(tf.random_normal([CONV2_f, CONV2_f, CONV1_n, CONV2_n]))
        self.bconv2 = tf.Variable(tf.random_normal([CONV2_n]))
        
        self.Wfull1 = tf.Variable(tf.random_normal([13*13*16, FULL1_n]))
        self.bfull1 = tf.Variable(tf.random_normal([FULL1_n]))
        
        self.Wfull2 = tf.Variable(tf.random_normal([FULL1_n, FULL2_n]))
        self.bfull2 = tf.Variable(tf.random_normal([FULL2_n]))
        
        self.WOut = tf.Variable(tf.random_normal([FULL2_n, 1]))
        self.bOut = tf.Variable(tf.random_normal([1]))
              
        # Setup
        self.__setup_layers(self.input)
        self.__setup_training()
        self.__setup_model_evaluation()
        
        self.init = tf.global_variables_initializer()
        
    def __conv2d(self, input, W, b, strides=1):
        input = tf.nn.conv2d(input, W, strides=[1, strides, strides, 1], padding='VALID')
        input = tf.nn.bias_add(input, b)
        return tf.nn.relu(input)
    
    def __maxpool2d(self, input, k=2):
        return tf.nn.max_pool(input, ksize=[1, k, k, 1], strides=[1, k, k, 1], padding='VALID')
    
    def __full_conv(self, input, W, b):
        input = tf.add(tf.matmul(input, W), b)
        input = tf.nn.relu(input);
        return input
    
    def __setup_layers(self, input):
        #input = tf.reshape(input, shape=[-1, self.image_height, self.image_width, self.image_channels])
        
        # First Layer
        conv1 = self.__conv2d(input, self.Wconv1, self.bconv1)
        conv1 = self.__maxpool2d(conv1)
        
        # Second Layer
        conv2 = self.__conv2d(conv1, self.Wconv2, self.bconv2)
        conv2 = self.__maxpool2d(conv2)
        
        # Reshape
        reshaped = tf.reshape(conv2, [self.batch_size, -1])
        
        # Full Layers
        full1 = self.__full_conv(reshaped, self.Wfull1, self.bfull1)
        full2 = self.__full_conv(full1, self.Wfull2, self.bfull2)
        
        # Out
        self.logits = tf.add(tf.matmul(full2, self.WOut), self.bOut)
        self.prediction = tf.nn.sigmoid(self.logits);
        
    def __setup_training(self):
        self.loss = tf.reduce_mean(tf.nn.sigmoid_cross_entropy_with_logits(logits=self.logits, labels=self.output))
        self.optimizer = tf.train.AdamOptimizer(learning_rate=self.learning_rate)
        self.train_step = self.optimizer.minimize(self.loss)
        
    def __setup_model_evaluation(self):
        equal = tf.equal(
            tf.cast(self.output[0:self.num], tf.bool),
            tf.cast(tf.round(self.prediction[0:self.num]), tf.bool),
        )
        self.num_correct = tf.reduce_sum(tf.cast(equal, tf.float32), name='accuracy_computation')

Adapt your CNN model above, so it can also do classification for more than two classes. Add if-else blocks to your model where appropriate. For more than two classes you will need the following:

  1. Your output placeholder will be of shape batch_size x num_classes (one-hot encoding).
  2. The dimensionality of your output layer should not be 1 but num_classes (affects the size of the last weight matrix and the last biases).

  4. For the predictions, use tf.nn.softmax.

class CNN2:
    def __init__(self, num_classes, image_height, image_width, image_channels, batch_size, learning_rate):
        
        # Parameters
        self.batch_size = batch_size
        self.learning_rate = learning_rate
        self.image_height = image_height
        self.image_width = image_width
        self.image_channels = image_channels
        
        # Inputs
        self.num = tf.placeholder(tf.int32)
        self.input = tf.placeholder(tf.float32, shape=[batch_size, image_height, image_width, image_channels])
        self.output = tf.placeholder(tf.float32, shape=[None, num_classes])
        
        # Weights & biases
        self.Wconv1 = tf.Variable(tf.random_normal([CONV1_f, CONV1_f, image_channels, CONV1_n]))
        self.bconv1 = tf.Variable(tf.random_normal([CONV1_n]))
        
        self.Wconv2 = tf.Variable(tf.random_normal([CONV2_f, CONV2_f, CONV1_n, CONV2_n]))
        self.bconv2 = tf.Variable(tf.random_normal([CONV2_n]))
        
        self.Wfull1 = tf.Variable(tf.random_normal([4*4*16, FULL1_n]))
        self.bfull1 = tf.Variable(tf.random_normal([FULL1_n]))
        
        self.Wfull2 = tf.Variable(tf.random_normal([FULL1_n, FULL2_n]))
        self.bfull2 = tf.Variable(tf.random_normal([FULL2_n]))
        
        self.WOut = tf.Variable(tf.random_normal([FULL2_n, num_classes]))
        self.bOut = tf.Variable(tf.random_normal([num_classes]))
              
        # Setup
        self.__setup_layers(self.input)
        self.__setup_training()
        self.__setup_model_evaluation()
        
        self.init = tf.global_variables_initializer()
        
    def __conv2d(self, input, W, b, strides=1):
        input = tf.nn.conv2d(input, W, strides=[1, strides, strides, 1], padding='VALID')
        input = tf.nn.bias_add(input, b)
        return tf.nn.relu(input)
    
    def __maxpool2d(self, input, k=2):
        return tf.nn.max_pool(input, ksize=[1, k, k, 1], strides=[1, k, k, 1], padding='VALID')
    
    def __full_conv(self, input, W, b):
        input = tf.add(tf.matmul(input, W), b)
        input = tf.nn.relu(input);
        return input
    
    def __setup_layers(self, input):
        #input = tf.reshape(input, shape=[-1, self.image_height, self.image_width, self.image_channels])
        
        # First Layer
        conv1 = self.__conv2d(input, self.Wconv1, self.bconv1)
        conv1 = self.__maxpool2d(conv1)
        
        # Second Layer
        conv2 = self.__conv2d(conv1, self.Wconv2, self.bconv2)
        conv2 = self.__maxpool2d(conv2)
        
        # Reshape
        reshaped = tf.reshape(conv2, [self.batch_size, -1])
        
        # Full Layers
        full1 = self.__full_conv(reshaped, self.Wfull1, self.bfull1)
        full2 = self.__full_conv(full1, self.Wfull2, self.bfull2)
        
        # Out
        self.logits = tf.add(tf.matmul(full2, self.WOut), self.bOut)
        self.prediction = tf.nn.softmax(self.logits);
        
    def __setup_training(self):
        self.loss = tf.reduce_mean(tf.nn.softmax_cross_entropy_with_logits(logits=self.logits, labels=self.output))
        self.optimizer = tf.train.AdamOptimizer(learning_rate=self.learning_rate)
        self.train_step = self.optimizer.minimize(self.loss)
        
    def __setup_model_evaluation(self):
        equal = tf.equal(
            tf.argmax(self.prediction[0:self.num], 1),
            tf.argmax(self.output[0:self.num], 1),
        )
        self.num_correct = tf.reduce_sum(tf.cast(equal, tf.float32), name='accuracy_computation')

    def dense_to_onehot(self, dense, num_classes):
        n = dense.shape[0]
        result = np.zeros((n, num_classes), dtype=np.int8)
        index_offset = np.arange(n) * num_classes
        result.flat[index_offset + dense.ravel()] = 1
        return result

    def training_batches(self, batch_size):
        shuf = np.arange(len(self.train_images))
        np.random.shuffle(shuf)
        images = self.train_images[shuf]
        labels = self.train_labels[shuf]
        offset = 0
        while offset < len(images):
            yield (images[offset:offset+batch_size], labels[offset:offset+batch_size])
            offset += batch_size

    def test_batches(self, batch_size):
        shuf = np.arange(len(self.test_images))

        images = self.test_images[shuf]
        labels = self.test_labels[shuf]
        offset = 0
        while offset < len(images):
            yield (images[offset:offset+batch_size], labels[offset:offset+batch_size])
            offset += batch_size
